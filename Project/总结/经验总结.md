           int partition = partition(arr, left, right);

            quickSort(arr,partition+1,right);

    public static int partition(int[] arr,int left,int right){

                swap(arr,i,index);
                index++;
            }
        }
        swap(arr,pivot,index-1);
        return index-1;
    }
